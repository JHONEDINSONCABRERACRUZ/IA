##Funcion Normalizacin - escalar datos entre 0 y 1
norm_minmax <- function(x){
  (x- min(x)) /(max(x)-min(x))
}

##Funcion para denormalizar
denorm_minmax <- function(x_norm, min_orig, max_orig){
  x_denorm <- x_norm * (max_orig - min_orig) + min_orig
  return(x_denorm)
}

ShapiroFuncion <- function(DFcol, nom){
  resultado<-shapiro.test(DFcol)
  if (resultado$p.value < 0.05){

    print(paste('la variable es normal:',nom, 'Valor de P:', resultado$p.value))
  }
}
